CREATE TABLE Client (
"CID" NUMBER(4,0) NOT NULL ENABLE,
"FName" VARCHAR2(15),
"LName" VARCHAR2(15),
"DOB" DATE,
"Address" VARCHAR2(30),
"City" VARCHAR2(15),
"County" VARCHAR2(15),
"Email" VARCHAR2(30),
"Mobile" VARCHAR2(20),
"Notes" VARCHAR2(50),
PRIMARY KEY (CID)
)

CREATE TABLE Physio (
"PHYID" NUMBER(2,0) NOT NULL ENABLE,
"FName" VARCHAR2(15),
"LName" VARCHAR2(15),
"HireDate" DATE,
"Qual" VARCHAR2(30),
"Salary" NUMBER(9,2),
"PhyRole" VARCHAR2(15),
PRIMARY KEY (PHYID)
)
